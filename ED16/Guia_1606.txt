; Quadrado de um numero (soma de impares)

    .ORG 8000H

    LXI H, 0000H    ; Zera o acumulador (resultado final)
    LDA 0100H       ; Carrega o numero N
    MOV C, A        ; C = contador
    
    MVI E, 01H      ; Começa com o primeiro impar (1)
    MVI D, 00H      ; Zera parte alta de DE

LOOP:
    DAD D           ; Soma o impar atual no total (HL)
    
    ; Calcula proximo impar (+2)
    MOV A, E
    ADI 02H
    MOV E, A        ; Soma 2 na parte baixa
    JNC SKIP        ; Se não houver carry
    
    INR D           ; Se deu carry, incrementa a parte alta

SKIP:
    DCR C           ; Conta uma volta
    JNZ LOOP        ; Repete N vezes

    SHLD 0102H      ; Guarda resultado de 16 bits
    HLT
