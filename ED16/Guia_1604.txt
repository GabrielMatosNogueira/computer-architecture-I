; Encontrar o menor valor do vetor

    .ORG 8000H

    LDA 0100H    ; Carrega o tamanho do vetor
    MOV C, A     ; C = contador do loop
    DCR C        ; Decrementa 1 (n-1)
    
    LXI H, 0101H ; HL aponta para o inicio dos dados
    MOV A, M     ; Inicializa o 'min' com o primeiro valor

LOOP:
    INX H        ; Aponta para o proximo dado
    CMP M        ; Compara A com a Memoria
    JC SKIP      ; Se A < M (Carry), pula a troca
    
    MOV A, M     ; Se nao, atualiza o novo menor

SKIP:
    DCR C        ; Decrementa o contador
    JNZ LOOP     ; Se C != 0, repete o loop

    STA 0105H    ; Armazena o resultado final
    HLT
