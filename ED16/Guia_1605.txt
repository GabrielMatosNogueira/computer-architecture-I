; Conversao BCD para Hex

    .ORG 8000H

    LDA 0100H    ; Pega o valor BCD
    MOV B, A     ; Salva backup em B

    ANI F0H      ; Isola a parte alta
    RRC
    RRC
    RRC
    RRC          ; Shift para mover a dezena para o inicio

    ; Multiplica por 10 (usando x2 + x8)
    ADD A        ; x2
    MOV E, A     ; Guarda no E
    ADD A        ;
    ADD A        ; 
    ADD E        ; 
    
    MOV D, A     ; D 

    MOV A, B     ; Recupera o valor original
    ANI 0FH      ; Limpa a parte alta, deixa so a unidade

    ADD D        ; Soma unidade + dezena convertida

    STA 0101H    ; Guarda resultado
    HLT
