; Calculo de Fatorial (N!) com subrotina

    .ORG 8000H
    JMP MAIN

MULT:
    PUSH B       ; Salva os registradores para nao estragar o loop principal
    PUSH D
    
    MOV D, H     ; Copia o valor atual (HL) para DE
    MOV E, L     ; porque o DAD soma DE no HL
    
    LXI H, 0000H ; Zera o HL para usar como acumulador da soma
    
MLOOP:
    DAD D        ; Soma o valor original
    DCR C        ; Decrementa contador da multiplicacao
    JNZ MLOOP    ; Repete ate somar C vezes
    
    POP D        ; Recupera os valores antigos
    POP B
    RET

MAIN:
    LDA 0100H    ; Carrega o N
    MOV C, A     ; C controla o fatorial
    
    LXI H, 0001H ; Comeca com 1

LOOP:
    MOV A, C
    CPI 01H      ; Ve se ja chegou no 1
    JZ DONE      ; Se sim, acabou
    
    CALL MULT    ; Faz HL = HL * C
    
    DCR C        ; Decrementa o numero do fatorial
    JMP LOOP     ; Volta pro loop

DONE:
    SHLD 0102H   ; Salva o resultado final (16 bits)
    HLT
